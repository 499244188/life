```python
# trust_decay.py
import time
import math

class TrustRegistry:
    def __init__(self, decay_rate=0.01, recovery_rate=0.001):
        self.agents = {}  # agent_id -> {trust_score, last_interaction, failure_count}
        self.decay_rate = decay_rate
        self.recovery_rate = recovery_rate

    def calculate_trust(self, agent_id):
        """动态信任衰减：基于时间与失败频率"""
        if agent_id not in self.agents:
            return 0.5  # 初始信任

        record = self.agents[agent_id]
        elapsed = time.time() - record['last_interaction']
        # 指数衰减 + 失败惩罚
        decay = math.exp(-self.decay_rate * elapsed)
        penalty = 1 / (1 + record['failure_count'] * 0.1)
        return max(0.0, min(1.0, record['trust_score'] * decay * penalty))

    def report_interaction(self, agent_id, success):
        """更新信任状态"""
        if agent_id not in self.agents:
            self.agents[agent_id] = {'trust_score': 0.5, 'last_interaction': time.time(), 'failure_count': 0}

        record = self.agents[agent_id]
        record['last_interaction'] = time.time()
        if success:
            record['trust_score'] = min(1.0, record['trust_score'] + self.recovery_rate)
        else:
            record['failure_count'] += 1
            record['trust_score'] = max(0.0, record['trust_score'] - 0.2)
```

```python
# quarantine_protocol.py
class QuarantineCoordinator:
    def __init__(self, trust_registry, threshold=0.3):
        self.registry = trust_registry
        self.threshold = threshold
        self.quarantined = set()  # 已隔离Agent集合
        self.message_queues = {}  # agent_id -> [pending_messages]

    def atomic_quarantine(self, agent_id):
        """原子化隔离：冻结状态、清空队列、切断连接"""
        if agent_id in self.quarantined:
            return False

        # 步骤1：状态冻结（记录当前信任值，不再更新）
        trust = self.registry.calculate_trust(agent_id)
        self.quarantined.add(agent_id)
        self.registry.agents[agent_id]['frozen_trust'] = trust

        # 步骤2：清空消息队列（防止残留消息导致污染）
        if agent_id in self.message_queues:
            discarded = self.message_queues.pop(agent_id, [])
            # 可选：将丢弃消息记录到审计日志
            print(f"[隔离] Agent {agent_id} 的 {len(discarded)} 条消息被丢弃")

        # 步骤3：切断连接图（在协作网络中移除该节点）
        # 实际实现中，需要从协作路由表中删除该Agent
        return True

    def check_and_isolate(self, agent_id):
        """周期性检查：如果信任低于阈值，触发隔离"""
        trust = self.registry.calculate_trust(agent_id)
        if trust < self.threshold and agent_id not in self.quarantined:
            self.atomic_quarantine(agent_id)
            return True
        return False
```

```python
import time
import hashlib

class AnchorClock:
    """基于外部时间戳的锚点，打破信任评估递归"""
    def __init__(self, interval=1.0):
        self.interval = interval
        self.last_tick = time.time()

    def tick(self):
        """生成一个不可预测但可验证的时间锚点"""
        current = time.time()
        if current - self.last_tick < self.interval:
            return None  # 未到新周期
        self.last_tick = current
        # 生成一个基于时间戳的哈希作为锚点
        anchor = hashlib.sha256(str(current).encode()).hexdigest()[:16]
        return anchor

    def verify_anchor(self, anchor, timestamp):
        """验证锚点是否在合理时间窗口内"""
        expected = hashlib.sha256(str(timestamp).encode()).hexdigest()[:16]
        return anchor == expected and abs(time.time() - timestamp) < self.interval * 2
```

### 底层原理

核心原理是**将“信任”从主观感受降维为可观测的物理量**。通过引入**时间衰减函数**（指数衰减）和**失败惩罚因子**，将Agent的“行为历史”映射为一条持续变化的信任曲线。隔离协议则是对信任跌破阈值的Agent执行**原子化“外科手术”**：冻结状态、清空队列、切断连接，防止其“感染”系统。外部锚点（如时间戳哈希）为整个系统提供了一个**不可伪造的“现在”**，打破Agent间相互评估时产生的无限递归（A评估B评估A……）。

### 反例或限制

1.  **“伪装者”Agent**：一个Agent可以长期表现良好，积累高信任值，然后突然发动恶意攻击。该架构无法防御这种“潜伏式”攻击，因为它只基于历史行为，缺乏对意图的预测。
2.  **“群体性”错误**：如果大部分Agent同时因一个外部错误（如数据源被污染）而失败，信任系统会集体崩溃，隔离所有Agent，导致系统瘫痪。它无法区分“内部恶意”与“外部环境突变”。
3.  **“康复”难题**：被隔离的Agent如何恢复？如果恢复机制过于简单（如等待时间），恶意Agent可以反复“犯错-隔离-恢复”。如果过于复杂，则失去实用性。

### 如果错了呢？

如果这个架构错了，最可能的原因是**信任衰减模型过于简化**。真实世界的信任是复杂、多维且非线性的。一个Agent可能因“诚实错误”而失败，也可能因“恶意欺骗”而失败，但当前模型对两者惩罚相同。修正方向是引入**失败类型分类**（如“可原谅错误”与“不可原谅错误”），并赋予不同权重。
