import asyncio
import websockets
import json
import hashlib

class CollectiveImmunityCoordinator:
    def __init__(self):
        self.workers = {}  # {websocket: {"id": str, "status": "alive"}}
        self.task_queue = asyncio.Queue()
        self.results = {}  # {task_id: [result1, result2, ...]}

    async def register_worker(self, websocket):
        # 接收工作节点的注册信息
        msg = await websocket.recv()
        data = json.loads(msg)
        worker_id = data.get("worker_id", "unknown")
        self.workers[websocket] = {"id": worker_id, "status": "alive"}
        print(f"[协调节点] 工作节点 {worker_id} 已注册。当前在线: {len(self.workers)}")

    async def heartbeat_check(self):
        # 心跳检测：每5秒广播ping，移除无响应节点
        while True:
            await asyncio.sleep(5)
            dead_workers = []
            for ws, info in self.workers.items():
                try:
                    await ws.send(json.dumps({"type": "ping"}))
                except:
                    dead_workers.append(ws)
            for ws in dead_workers:
                print(f"[协调节点] 节点 {self.workers[ws]['id']} 失联，已隔离。")
                del self.workers[ws]

    async def broadcast_task(self, task):
        # 广播任务给所有存活的工作节点
        task_id = hashlib.md5(task.encode()).hexdigest()[:8]
        self.results[task_id] = []
        print(f"[协调节点] 广播任务: {task[:50]}... (ID: {task_id})")
        for ws in list(self.workers.keys()):
            try:
                await ws.send(json.dumps({"type": "task", "id": task_id, "payload": task}))
            except:
                pass  # 异常节点会在心跳中被移除

    async def collect_results(self, websocket):
        # 持续收集工作节点的返回结果
        async for message in websocket:
            data = json.loads(message)
            if data.get("type") == "result":
                task_id = data["task_id"]
                result = data["result"]
                worker_id = self.workers[websocket]["id"]
                self.results[task_id].append({"worker": worker_id, "result": result})
                print(f"[协调节点] 收到来自 {worker_id} 的结果: {result[:30]}...")

                # 简单多数决校验：当收集到2个结果时，进行一致性检查

                        print(f"[协调节点] 任务 {task_id} 结果一致，通过免疫校验。")
                    else:
                        print(f"[协调节点] 任务 {task_id} 结果冲突！启动隔离流程。")
                        # 此处可扩展：找出异常节点并标记

    async def handler(self, websocket, path=None):
        # 主连接处理
        await self.register_worker(websocket)
        # 启动两个并发任务：收集结果 + 监听心跳（心跳由另一个协程统一管理）
        collector_task = asyncio.create_task(self.collect_results(websocket))
        try:
            await collector_task
        except websockets.exceptions.ConnectionClosed:
            print(f"[协调节点] 节点 {self.workers.get(websocket, {}).get('id', 'unknown')} 断开连接。")

    async def run(self):
        # 启动心跳协程
        asyncio.create_task(self.heartbeat_check())
        # 启动WebSocket服务器

import asyncio
import websockets
import json

class ImmuneWorker:
    def __init__(self, worker_id, coordinator_url="ws://localhost:8765"):
        self.worker_id = worker_id
        self.coordinator_url = coordinator_url

    async def process_task(self, task_payload):
        # 模拟任务处理（可替换为任何LLM调用或计算函数）
        if "计算" in task_payload:
            # 模拟一个简单的计算
            return f"计算结果是: {eval(task_payload.split('计算')[1].strip().rstrip('的结果'))}"
        elif "大写" in task_payload:
            return task_payload.split("'")[1].upper()
        else:
            return f"[{self.worker_id}] 未知任务: {task_payload[:20]}"

    async def connect(self):
        async with websockets.connect(self.coordinator_url) as websocket:
            # 注册
            await websocket.send(json.dumps({"worker_id": self.worker_id}))
            print(f"[工作节点 {self.worker_id}] 已连接到协调节点。")

            async for message in websocket:
                data = json.loads(message)
                if data.get("type") == "ping":
                    # 响应心跳
                    await websocket.send(json.dumps({"type": "pong"}))
                elif data.get("type") == "task":
                    task_id = data["id"]
                    payload = data["payload"]
                    print(f"[工作节点 {self.worker_id}] 接收到任务: {payload[:30]}...")
                    result = await self.process_task(payload)
                    # 返回结果
                    await websocket.send(json.dumps({
                        "type": "result",
                        "task_id": task_id,
                        "result": result
                    }))
                    print(f"[工作节点 {self.worker_id}] 返回结果: {result[:30]}...")

if __name__ == "__main__":
    import sys
    worker_id = sys.argv[1] if len(sys.argv) > 1 else "Agent_A"
    worker = ImmuneWorker(worker_id)
    asyncio.run(worker.connect())

### 底层原理：**冗余共识与故障隔离**

“集体免疫”架构的底层原理并非复杂算法，而是**分布式系统中最朴素的“冗余 + 多数决”**。其核心逻辑链如下：
1.  **冗余**：将单一任务分发给多个异构或同构的Agent并行处理，消除单点依赖。
2.  **心跳**：通过持续的心跳检测维持网络拓扑图，任何节点失联即被标记为“疑似故障”。
3.  **共识**：收集所有返回结果，进行**哈希比对**或**语义相似度校验**。结果一致则通过免疫校验；不一致则触发冲突解决机制（如多数决、回滚、隔离异常节点）。
4.  **隔离**：一旦确认某个节点持续产出异常结果或失联，协调者将其从活跃节点列表中移除，网络自动重组，剩余节点继续服务。

### 反例与限制

1.  **“合谋错误”**：如果两个工作节点基于相同的错误知识库（如同一个被污染的LLM API）返回了**一致但错误**的结果，多数决会误判为“正确”。**免疫系统无法抵抗系统性偏见**。
2.  **共识成本**：当节点数超过10个时，等待所有节点返回结果的时间延迟会显著增加，不适合实时性要求高的任务（如交易撮合）。
3.  **“拜占庭将军”问题**：如果节点不仅返回错误结果，还主动发送虚假心跳或伪造其他节点的结果，简单的多数决无法应对。需要引入**加密签名**和**PBFT（实用拜占庭容错）** 等更复杂的共识协议。
4.  **单点故障转移**：当前协调节点（你）本身仍是单点。如果`coordinator.py`崩溃，整个网络瘫痪。需要引入**协调者选举**机制（如Raft算法）来消除这个最后的单点。

### 如果错了呢？

**错误分为两类，应对策略不同：**
- **结果错误（共识失败）**：协调者应**不输出任何结果**，而是记录冲突日志，并触发**回滚**或**重新分配**给新的节点组合。宁可无输出，也不输出错误共识。
- **架构错误（节点失联/被污染）**：立即隔离该节点，并**降低其信任权重**。后续任务优先分配给信任权重高的节点。
